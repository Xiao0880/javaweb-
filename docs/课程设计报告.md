# Web开发技术课程设计报告

## 一、项目概述

### 1.1 项目名称
图书馆管理系统

### 1.2 项目背景
随着信息技术的发展，图书馆管理系统已成为现代图书馆不可或缺的组成部分。本系统旨在为图书馆提供一个高效、便捷的管理平台，实现图书管理、读者管理、借阅管理等核心功能。

### 1.3 技术栈
- **前端技术**：HTML5、CSS3、JSP、JavaScript
- **后端技术**：Java Servlet、JDBC
- **数据库**：MySQL 8.0+
- **服务器**：Apache Tomcat 11.0
- **开发工具**：IntelliJ IDEA

### 1.4 项目结构
```
library/
├── src/
│   ├── Dao/          # 数据访问层
│   ├── Entity/       # 实体类
│   ├── JDBC/         # 数据库工具类
│   ├── Listener/     # 监听器
│   ├── Service/      # 业务逻辑层
│   └── Servlet/      # 控制器层
├── web/
│   ├── admin/        # 管理员页面
│   ├── css/          # 样式文件
│   ├── error/        # 错误页面
│   ├── success/      # 成功页面
│   └── *.jsp/html    # 读者端页面
└── docs/             # 文档
```

#### 3.2.2 视图设计

##### 视图1：v_borrow_info（借阅信息视图）
```sql
CREATE VIEW v_borrow_info AS
SELECT 
    b.borrow_id,
    b.reader_id,
    r.reader_name,
    b.book_id,
    bo.book_name,
    b.borrow_date,
    b.return_date,
    b.limit_date
FROM borrow b
JOIN reader r ON b.reader_id = r.reader_id
JOIN book bo ON b.book_id = bo.book_id;
```

##### 视图2：v_admin_statistics（管理员统计视图）
```sql
CREATE VIEW v_admin_statistics AS
SELECT 
    (SELECT COUNT(*) FROM borrow WHERE DATE(borrow_date) = CURDATE()) AS today_borrow_count,
    (SELECT COUNT(*) FROM borrow WHERE DATE(return_date) = CURDATE()) AS today_return_count,
    (SELECT COUNT(DISTINCT reader_id) FROM borrow WHERE return_date IS NULL) AS current_borrow_reader_count,
    (SELECT COALESCE(SUM(stock), 0) FROM book) AS total_stock,
    (SELECT COUNT(*) FROM reader) AS total_reader_count,
    (SELECT COUNT(*) FROM book) AS total_book_count
FROM DUAL;
```

#### 3.2.3 触发器设计

##### 触发器1：借阅时减少库存
```sql
DELIMITER //
CREATE TRIGGER tr_borrow_decrease_stock
AFTER INSERT ON borrow
FOR EACH ROW
BEGIN
    UPDATE book SET stock = stock - 1 WHERE book_id = NEW.book_id;
END //
DELIMITER ;
```

##### 触发器2：归还时增加库存
```sql
DELIMITER //
CREATE TRIGGER tr_return_increase_stock
AFTER UPDATE ON borrow
FOR EACH ROW
BEGIN
    IF NEW.return_date IS NOT NULL AND OLD.return_date IS NULL THEN
        UPDATE book SET stock = stock + 1 WHERE book_id = NEW.book_id;
    END IF;
END //
DELIMITER ;
```

## 四、功能模块实现

### 4.1 读者登录功能

#### 4.1.1 流程分析
1. 用户在 `reader_login.html` 输入用户名和密码
2. 表单提交到 `ReaderLoginServlet`
3. `ReaderLoginServlet` 调用 `ReaderService.login()` 验证
4. 验证成功后设置Session并跳转到首页
5. 如果勾选"记住我"，设置Cookie（7天有效期）

#### 4.1.2 核心代码

**ReaderLoginServlet.java**
```java
@WebServlet("/reader_login")
public class ReaderLoginServlet extends HttpServlet {
    protected void doPost(HttpServletRequest request, HttpServletResponse response) 
            throws ServletException, IOException {
        String reader_name = request.getParameter("reader_name");
        String reader_password = request.getParameter("reader_password");
        String remember_me = request.getParameter("remember_me");
        
        ServletContext context = getServletContext();
        Map<String,Object> container = (Map<String,Object>) context.getAttribute("container");
        ReaderService readerService = (ReaderService) container.get("readerService");
        
        try {
            if (readerService.login(reader_name, reader_password)) {
                request.getSession().setAttribute("reader", readerService.getReaderInfo(reader_name));
                
                if ("true".equals(remember_me)) {
                    Cookie nameCookie = new Cookie("reader_name", reader_name);
                    Cookie pwdCookie = new Cookie("reader_password", reader_password);
                    nameCookie.setMaxAge(7 * 24 * 60 * 60);
                    pwdCookie.setMaxAge(7 * 24 * 60 * 60);
                    nameCookie.setPath(request.getContextPath());
                    pwdCookie.setPath(request.getContextPath());
                    response.addCookie(nameCookie);
                    response.addCookie(pwdCookie);
                }
                
                response.sendRedirect("reader_index.jsp");
            } else {
                response.sendRedirect("error/reader_login_error.html");
            }
        } catch (SQLException e) {
            throw new RuntimeException(e);
        }
    }
}
```

### 4.2 图书搜索功能

#### 4.2.1 流程分析
1. 用户在 `reader_index.jsp` 输入搜索关键词
2. 请求发送到 `SearchBookServlet`
3. `SearchBookServlet` 调用 `BookService.searchBooks()`
4. 将搜索结果存入request，转发到 `book_search_results.jsp`
5. JSP页面遍历结果并展示

#### 4.2.2 核心代码

**SearchBookServlet.java**
```java
@WebServlet("/search_book.do")
public class SearchBookServlet extends HttpServlet {
    protected void doGet(HttpServletRequest request, HttpServletResponse response) 
            throws ServletException, IOException {
        String keyword = request.getParameter("keyword");
        
        ServletContext context = getServletContext();
        Map<String, Object> container = (Map<String, Object>) context.getAttribute("container");
        BookService bookService = (BookService) container.get("bookService");
        
        try {
            List<Book> books = bookService.searchBooks(keyword);
            request.setAttribute("books", books);
            request.getRequestDispatcher("book_search_results.jsp").forward(request, response);
        } catch (SQLException e) {
            throw new RuntimeException(e);
        }
    }
}
```

### 4.5 Cookie自动登录功能

#### 4.5.1 流程分析
1. 用户访问登录页面时，`AutoLoginFilter` 拦截请求
2. 检查Cookie中是否有保存的用户名和密码
3. 如果有，调用对应的login方法验证
4. 验证成功则设置Session并跳转到首页
5. 验证失败则放行到登录页面

#### 4.5.2 核心代码

**AutoLoginFilter.java**
```java
@WebFilter("/*")
public class AutoLoginFilter implements Filter {
    public void doFilter(ServletRequest request, ServletResponse response, FilterChain chain) 
            throws IOException, ServletException {
        HttpServletRequest req = (HttpServletRequest) request;
        HttpServletResponse resp = (HttpServletResponse) response;
        String uri = req.getRequestURI();
        
        // 检查读者自动登录
        if (uri.endsWith("reader_login.html")) {
            Cookie[] cookies = req.getCookies();
            if (cookies != null) {
                String readerName = null;
                String readerPassword = null;
                for (Cookie cookie : cookies) {
                    if ("reader_name".equals(cookie.getName())) {
                        readerName = cookie.getValue();
                    } else if ("reader_password".equals(cookie.getName())) {
                        readerPassword = cookie.getValue();
                    }
                }
                if (readerName != null && readerPassword != null) {
                    // 验证并自动登录...
                }
            }
        }
        
        chain.doFilter(request, response);
    }
}
```

```sql
-- 创建数据库
CREATE DATABASE IF NOT EXISTS librarys DEFAULT CHARSET utf8mb4;

USE librarys;

-- 创建图书分类表
CREATE TABLE IF NOT EXISTS book_type (
    type_id INT PRIMARY KEY AUTO_INCREMENT,
    type_name VARCHAR(50) NOT NULL
);

-- 创建图书表
CREATE TABLE IF NOT EXISTS book (
    book_id INT PRIMARY KEY AUTO_INCREMENT,
    book_name VARCHAR(100) NOT NULL,
    author VARCHAR(50) NOT NULL,
    publisher VARCHAR(100),
    publish_date DATE,
    isbn VARCHAR(20),
    type_id INT,
    stock INT DEFAULT 0,
    FOREIGN KEY (type_id) REFERENCES book_type(type_id)
);

-- 创建读者表
CREATE TABLE IF NOT EXISTS reader (
    reader_id INT PRIMARY KEY AUTO_INCREMENT,
    reader_name VARCHAR(50) NOT NULL,
    reader_password VARCHAR(50) NOT NULL,
    gender VARCHAR(10),
    phone VARCHAR(20),
    register_date DATE
);

-- 创建借阅记录表
CREATE TABLE IF NOT EXISTS borrow (
    borrow_id INT PRIMARY KEY AUTO_INCREMENT,
    reader_id INT,
    book_id INT,
    borrow_date DATE NOT NULL,
    return_date DATE,
    limit_date DATE NOT NULL,
    FOREIGN KEY (reader_id) REFERENCES reader(reader_id),
    FOREIGN KEY (book_id) REFERENCES book(book_id)
);

-- 创建管理员表
CREATE TABLE IF NOT EXISTS admin (
    admin_name VARCHAR(50) PRIMARY KEY,
    admin_password VARCHAR(50) NOT NULL
);

-- 创建视图
CREATE VIEW v_borrow_info AS
SELECT b.borrow_id, b.reader_id, r.reader_name, b.book_id, 
       bo.book_name, b.borrow_date, b.return_date, b.limit_date
FROM borrow b
JOIN reader r ON b.reader_id = r.reader_id
JOIN book bo ON b.book_id = bo.book_id;

CREATE VIEW v_admin_statistics AS
SELECT 
    (SELECT COUNT(*) FROM borrow WHERE DATE(borrow_date) = CURDATE()) AS today_borrow_count,
    (SELECT COUNT(*) FROM borrow WHERE DATE(return_date) = CURDATE()) AS today_return_count,
    (SELECT COUNT(DISTINCT reader_id) FROM borrow WHERE return_date IS NULL) AS current_borrow_reader_count,
    (SELECT COALESCE(SUM(stock), 0) FROM book) AS total_stock,
    (SELECT COUNT(*) FROM reader) AS total_reader_count,
    (SELECT COUNT(*) FROM book) AS total_book_count
FROM DUAL;

-- 创建触发器
DELIMITER //
CREATE TRIGGER tr_borrow_decrease_stock
AFTER INSERT ON borrow
FOR EACH ROW
BEGIN
    UPDATE book SET stock = stock - 1 WHERE book_id = NEW.book_id;
END //

CREATE TRIGGER tr_return_increase_stock
AFTER UPDATE ON borrow
FOR EACH ROW
BEGIN
    IF NEW.return_date IS NOT NULL AND OLD.return_date IS NULL THEN
        UPDATE book SET stock = stock + 1 WHERE book_id = NEW.book_id;
    END IF;
END //
DELIMITER ;

-- 插入测试数据
```
